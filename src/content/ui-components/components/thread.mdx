用于显示和交互评论线程的组件。

## 安装

```bash
npx @tiptap/cli add thread
```

## 使用方法

```tsx
import { Thread, ThreadFloating } from '@/components/tiptap-ui/thread'
import { useEditor, EditorContent } from '@tiptap/react'
import { Comments } from '@tiptap-pro/extension-comments'

export default function MyEditor() {
  const editor = useEditor({
    extensions: [
      // 其他扩展
      Comments.configure({
        // 评论配置
      }),
    ],
  })

  return (
    <div>
      <EditorContent editor={editor} />
      <ThreadFloating editor={editor} />
    </div>
  )
}
```

## 属性

| 名称   | 类型           | 默认值 | 描述                      |
| ------ | -------------- | ------- | ------------------------ |
| editor | Editor \| null | null    | Tiptap 编辑器实例      |
