import { Callout } from '@/components/ui/Callout'

```js
interface PagesOptions {
  /**
   * 页面格式，决定页面尺寸和边距。
   * 可以是内置格式名称或自定义页面格式对象。
   * 内置格式：A4、A3、A5、Letter、Legal、Tabloid
   * @default 'A4'
   */
  pageFormat: 'A4' | 'A3' | 'A5' | 'Letter' | 'Legal' | 'Tabloid' | PageFormat

  /**
   * 页面页眉高度，单位像素。
   * @default 50
   */
  headerHeight?: number

  /**
   * 页面页脚高度，单位像素。
   * @default 50
   */
  footerHeight?: number

  /**
   * 页面间距，单位像素。
   * @default 50
   */
  pageGap?: number

  /**
   * 每页的页眉内容，可以是字符串（支持占位符）或函数。
   * @default ''
   */
  header?: string | ((page: number, total: number) => string)

  /**
   * 每页的页脚内容，可以是字符串（支持占位符）或函数。
   * @default ''
   */
  footer?: string | ((page: number, total: number) => string)

  /**
   * 页面间距区域的背景颜色。
   * @default undefined
   */
  pageGapBackground?: string

  /**
   * 页面格式变化时调用的回调函数。
   * 接收新的页面格式作为参数。
   * @default () => {}
   */
  onPageFormatChange?: (pageFormat: PageFormat) => void

  /**
   * 用于视觉缩放的初始缩放级别。限制在 [0.25, 4]。
   * @default 1
   */
  zoom?: number

  /**
   * 缩放级别变化时调用的回调函数。
   * 接收新的缩放级别作为参数。
   */
  onZoomChange?: (zoom: number) => void
}

interface PageFormat {
  name: string
  width: number    // 宽度，单位像素
  height: number   // 高度，单位像素
  margins: {
    top: number    // 上边距，单位像素
    right: number  // 右边距，单位像素
    bottom: number // 下边距，单位像素
    left: number   // 左边距，单位像素
  }
}
```

| 选项                 | 类型 / 值                | 默认值      | 描述                                                                                         |
| -------------------- | ---------------------- | ----------- | ------------------------------------------------------------------------------------------ |
| `pageFormat`         | `string \| PageFormat` | `'A4'`      | 内置格式名称或自定义页面格式对象                                                              |
| `headerHeight`       | `number` (px)          | `50`        | 页眉高度，单位像素                                                                          |
| `footerHeight`       | `number` (px)          | `50`        | 页脚高度，单位像素                                                                          |
| `pageGap`            | `number` (px)          | `50`        | 页面之间的间距，单位像素                                                                      |
| `header`             | `string \| fn`         | `''`        | 页眉内容；字符串（支持 `{page}`/`{total}`）或 `(pageNumber, totalPages) => string` |
| `footer`             | `string \| fn`         | `'{page}'`  | 页脚内容；字符串（支持 `{page}`/`{total}`）或 `(pageNumber, totalPages) => string` |
| `pageGapBackground`  | `string` (CSS color)   | `undefined` | 页面间距区域的背景颜色                                                                      |
| `onPageFormatChange` | `fn`                   | `() => {}`  | 页面格式变化时的回调；接收新的 PageFormat                                                     |
| `zoom`               | `number`               | `1`         | 初始缩放级别（0.25–4）。参见 [缩放](/pages/core-concepts/zoom)                         |
| `onZoomChange`       | `fn`                   | `undefined` | 缩放变化时的回调；接收新的缩放级别                                                            |

## 示例用法

### 使用内置格式的基本配置

```js
import { Pages } from '@tiptap-pro/extension-pages'

const editor = new Editor({
  extensions: [
    Pages.configure({
      pageFormat: 'A4',
      headerHeight: 60,
      footerHeight: 40,
      pageGap: 40,
      header: '我的项目',
      footer: '第 {page} 页，共 {total} 页',
      pageGapBackground: '#f8f8f8',
    }),
  ],
})
```

### 自定义页面格式配置

```js
import { Pages } from '@tiptap-pro/extension-pages'

const editor = new Editor({
  extensions: [
    Pages.configure({
      pageFormat: {
        name: '自定义格式',
        width: 600,
        height: 800,
        margins: {
          top: 40,
          right: 30,
          bottom: 40,
          left: 30,
        },
      },
      header: '自定义文档',
      footer: '{page} / {total}',
      onPageFormatChange: (pageFormat) => {
        console.log('页面格式已更改:', pageFormat.name)
        // 保存用户偏好，更新界面等操作
      },
    }),
  ],
})
```

## 可用命令

Pages 扩展提供以下命令：

| 命令            | 参数                     | 描述                                                                               |
| --------------- | ------------------------ | ------------------------------------------------------------------------------------------ |
| `setPageFormat` | `pageFormat: PageFormat` | 以编程方式更改页面格式                                                                  |
| `setZoom`       | `zoom: number`           | 设置缩放级别（限制在 0.25–4）。参见 [缩放](/pages/core-concepts/zoom) |

### 命令用法

```js
// 切换到内置格式
editor.commands.setPageFormat('Letter')

// 切换到自定义格式
editor.commands.setPageFormat({
  id: 'wide-page-format',
  width: 1000, // 像素
  height: 700, // 像素
  margins: { top: 20, right: 20, bottom: 20, left: 20 } // 像素
})
```

<Callout title="注意">
  <code>header</code> 和 <code>footer</code> 选项可以接受字符串（带有 <code>\{page\}</code> 和 <code>\{total\}</code> 占位符）或函数 <code>(page, total) =&gt; string</code>，用于动态内容。
</Callout>

## 页眉和页脚编辑器存储

当页眉或页脚编辑器覆盖层打开时，Pages 扩展会通过 `editor.storage.pages` 暴露当前活动的内部编辑器。这对于自定义工具栏、状态 UI，以及监听页眉或页脚覆盖层中的编辑器事件非常有用。

| 存储属性           | 类型                           | 描述                                                            |
| ------------------ | ------------------------------ | ---------------------------------------------------------------------- |
| `activeEditor`     | `Editor \| null`               | 当前活动的页眉/页脚内部编辑器；如果没有活动覆盖层，则为 `null` |
| `activeEditorType` | `'header' \| 'footer' \| null` | 当前活动的内部编辑器属于页眉还是页脚                                |
| `activePageNumber` | `number \| null`               | 当前页眉/页脚编辑器对应的页码                                        |
| `headerEditorOn`   | `Editor['on'] \| null`         | 绑定到页眉内部编辑器的事件订阅辅助函数                                |
| `headerEditorOff`  | `Editor['off'] \| null`        | 绑定到页眉内部编辑器的事件取消订阅辅助函数                            |
| `footerEditorOn`   | `Editor['on'] \| null`         | 绑定到页脚内部编辑器的事件订阅辅助函数                                |
| `footerEditorOff`  | `Editor['off'] \| null`        | 绑定到页脚内部编辑器的事件取消订阅辅助函数                            |

`headerEditorOn` / `headerEditorOff` 和 `footerEditorOn` / `footerEditorOff` 辅助函数可让你订阅页眉和页脚编辑器的事件。这对于响应聚焦、选区更新或内容变更等事件非常有用。

```tsx
useEffect(() => {
  if (!editor) return

  const handleFocus = () => {
    const { activeEditorType, activePageNumber } = editor.storage.pages
    console.log('正在编辑', activeEditorType, activePageNumber)
  }

  editor.storage.pages.headerEditorOn?.('focus', handleFocus)
  editor.storage.pages.footerEditorOn?.('focus', handleFocus)

  return () => {
    editor.storage.pages.headerEditorOff?.('focus', handleFocus)
    editor.storage.pages.footerEditorOff?.('focus', handleFocus)
  }
}, [editor])
```

有关完整的工具栏示例，请参见 [页眉和页脚](/pages/core-concepts/page-header-footer#active-editor-state)。
